```
src/
├── app/
│   ├── config/          # Централизованная конфигурация
│   ├── core/            # Ядро приложения
│   │   ├── auth/        # Аутентификация и авторизация
│   │   ├── database/    # Работа с БД, транзакции, репликация
│   │   ├── cache/       # Распределенное кэширование
│   │   ├── storage/     # Работа с файловым хранилищем
│   │   ├── jobs/        # Очереди задач (RabbitMQ/Kafka)
│   │   ├── logging/     # Централизованное логирование
│   │   └── monitoring/  # Мониторинг и сбор метрик
│   └── services/        # Бизнес-логика, разделенная на сервисы
│       ├── case/        # Управление делами
│       ├── user/        # Управление пользователем
│       ├── client/      # Управление клиентами
│       ├── document/    # Управление документами
│       ├── finance/     # Финансовые операции
│       ├── calendar/    # Календарь и события
│       ├── mail/        # Почтовый сервис
│       ├── calculation/ # Расчеты по справочникам
│       ├── analytics/   # Аналитика и отчеты
│       └── notification/# Система уведомлений
├── infrastructure/
│   ├── persistence/     # Репозитории и работы с БД
│   ├── external/        # Интеграция с внешними сервисами
│   └── queue/           # Реализация очередей задач
└── interfaces/
    ├── http/            # REST API, контроллеры
    └── events/          # Обработка событий
```
